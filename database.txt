CREATE TABLE dbo.users (
user_id serial PRIMARY key,
username varchar(50) UNIQUE NOT NULL,
email varchar(100) UNIQUE NOT NULL,
password varchar(16) NOT NULL,
user_type varchar(10),
availableTime TIMESTAMP
);
